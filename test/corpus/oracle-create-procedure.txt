CREATE OR REPLACE PROCEDURE update_salary(
  p_emp_id IN NUMBER,
  p_pct    IN NUMBER
) AS
  v_old_sal NUMBER;
BEGIN
  UPDATE emp SET sal = v_old_sal WHERE empno = p_emp_id;
EXCEPTION
  WHEN NO_DATA_FOUND THEN
    RAISE;
END update_salary;
